jQuery InputFormat Demo

Input default

  • allowDecimals: true
  • allowNegative: true
  • allowLeadingZero: false
  • thousandSep: ,
  • decimalSep: .
  • maxDecimalDigits: unlimited

					$(document).ready(function(){
						$('#default').inputNumber();
					});

Input without Negative value

  • allowDecimals: true
  • allowNegative: false
  • allowLeadingZero: false
  • thousandSep: ,
  • decimalSep: .
  • maxDecimalDigits: unlimited

					$(document).ready(function(){
						$('#no-negative').inputNumber({ allowNegative: false });
					});

Input without Decimals value

  • allowDecimals: false
  • allowNegative: false
  • allowLeadingZero: false
  • thousandSep: ,
  • decimalSep: .
  • maxDecimalDigits: unlimited

					$(document).ready(function(){
						$('#no-decimal-negative').inputNumber({ allowDecimals: false, allowNegative: false });
					});

Max Decimals value 2 point

  • allowDecimals: true
  • allowNegative: true
  • allowLeadingZero: false
  • thousandSep: ,
  • decimalSep: .
  • maxDecimalDigits: 2

					$(document).ready(function(){
						$('#max-two-decimal').inputNumber({ maxDecimalDigits: 2 });
					});

Allow Leading Zero

  • allowDecimals: false
  • allowNegative: false
  • allowLeadingZero: true
  • thousandSep:
  • decimalSep: .
  • maxDecimalDigits: unlimited
  • numericOnly: true - this will override allowDecimals=false, allowNegative=false, thousandSep=empty string

					$(document).ready(function(){
						$('#allow-leading-zero').inputNumber({ allowLeadingZero: true, thousandSep: '', allowDecimals: false, allowNegative: false });
					});
					// Or using shortcut
					$(document).ready(function(){
						$('#allow-leading-zero').inputNumber({ allowLeadingZero: true, numericOnly: true });
					});
